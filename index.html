    <script>

        document.addEventListener('DOMContentLoaded', () => {

            const video = document.getElementById('camera-stream');

            const captureBtn = document.getElementById('capture-btn');

            const previewContainer = document.getElementById('preview-container');

            const capturedImage = document.getElementById('captured-image');

            const downloadBtn = document.getElementById('download-btn');

            const retakeBtn = document.getElementById('retake-btn');

            const loadingIndicator = document.getElementById('loading-indicator');

            const generateCaptionBtn = document.getElementById('generate-caption-btn');

            const captionContainer = document.getElementById('caption-container');

            const captionResult = document.getElementById('caption-result');

            const copyCaptionBtn = document.getElementById('copy-caption-btn');

            const copyHelper = document.getElementById('copy-helper');


            let stream = null;


            async function startCamera() {

                try {

                    stream = await navigator.mediaDevices.getUserMedia({

                        video: { facingMode: 'user', width: { ideal: 1280 }, height: { ideal: 720 } },

                        audio: false

                    });

                    video.srcObject = stream;

                } catch (err) {

                    console.error("Erro ao aceder à câmara: ", err);

                    alert("Não foi possível aceder à câmara. Verifique se deu permissão no seu navegador. Em iOS e Mac, é necessário usar uma ligação segura (https://).");

                }

            }


            function takePicture() {

                loadingIndicator.classList.remove('hidden');


                const compositionCanvas = document.getElementById('composition-canvas');

                const context = compositionCanvas.getContext('2d');

                const frameOverlay = document.getElementById('frame-overlay');


                compositionCanvas.width = video.videoWidth;

                compositionCanvas.height = video.videoHeight;


                context.drawImage(video, 0, 0, compositionCanvas.width, compositionCanvas.height);


                const frameImage = new Image();

                // Como a imagem está no mesmo servidor (GitHub Pages), não precisamos de 'crossOrigin'

                frameImage.src = frameOverlay.src;


                frameImage.onload = () => {

                    context.drawImage(frameImage, 0, 0, compositionCanvas.width, compositionCanvas.height);

                    const imageDataUrl = compositionCanvas.toDataURL('image/png');

                    capturedImage.src = imageDataUrl;

                    downloadBtn.href = imageDataUrl;

                    loadingIndicator.classList.add('hidden');

                    previewContainer.classList.add('visible');

                    captureBtn.classList.add('hidden');

                    captionContainer.classList.remove('hidden');

                };


                frameImage.onerror = () => {

                    loadingIndicator.classList.add('hidden');

                    alert('ERRO CRÍTICO: Não foi possível carregar o ficheiro "moldura.png". Verifique se o nome do ficheiro está correto e se foi carregado para o GitHub.');

                };

            }


            function retakePicture() {

                previewContainer.classList.remove('visible');

                captureBtn.classList.remove('hidden');

                captionContainer.classList.add('hidden');

                copyCaptionBtn.classList.add('hidden');

                captionResult.textContent = 'Clique em "Gerar Legenda" para que a IA crie uma mensagem para você!';

            }


            async function generateCaption() {

                captionResult.textContent = 'A gerar legenda com a IA...';

                generateCaptionBtn.disabled = true;

                copyCaptionBtn.classList.add('hidden');


                const prompt = "Crie 3 opções de legendas curtas e divertidas para uma foto de festa de aniversário de 1 ano. A moldura tem o tema dos Looney Tunes. As legendas devem ser em português do Brasil e otimistas. Formate a resposta como uma lista numerada (ex: 1. ... 2. ...).";


                const payload = { contents: [{ parts: [{ text: prompt }] }] };

                const apiKey = "";

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                try {

                    const response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });


                    if (!response.ok) throw new Error(`API request failed with status ${response.status}`);

                    

                    const result = await response.json();

                    

                    if (result.candidates && result.candidates[0]?.content?.parts[0]) {

                        const text = result.candidates[0].content.parts[0].text;

                        captionResult.innerHTML = text.replace(/\n/g, '<br>');

                        copyCaptionBtn.classList.remove('hidden');

                    } else {

                        captionResult.textContent = 'Não foi possível gerar uma legenda. Tente novamente.';

                    }

                } catch (error) {

                    console.error('Erro ao chamar a API Gemini:', error);

                    captionResult.textContent = 'Erro ao conectar com a IA. Tente novamente mais tarde.';

                } finally {

                    generateCaptionBtn.disabled = false;

                }

            }

            

            function copyCaption() {

                const textToCopy = captionResult.innerText;

                copyHelper.value = textToCopy;

                copyHelper.select();

                copyHelper.setSelectionRange(0, 99999);

                try {

                    document.execCommand('copy');

                    alert('Legenda copiada para a área de transferência!');

                } catch (err) {

                    alert('Não foi possível copiar a legenda.');

                }

            }


            captureBtn.addEventListener('click', takePicture);

            retakeBtn.addEventListener('click', retakePicture);

            generateCaptionBtn.addEventListener('click', generateCaption);

            copyCaptionBtn.addEventListener('click', copyCaption);


            startCamera();

        });
